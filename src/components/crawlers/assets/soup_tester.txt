var wpcom_remote_login_extra_auth = '';
function wpcom_remote_login_remove_dom_node_id( element_id ) {
	var dom_node = document.getElementById( element_id );
	if ( dom_node ) { dom_node.remove(); }
}
function wpcom_remote_login_remove_dom_node_classes( class_name ) {
	var dom_nodes = document.querySelectorAll( '.' + class_name );
	for ( var i = 0; i < dom_nodes.length; i++ ) {
		dom_nodes[i].remove();
	}
}
function wpcom_remote_login_final_cleanup() {
	wpcom_remote_login_remove_dom_node_classes( "wpcom_remote_login_msg" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_validate" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_js" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_request_access_iframe" );
	wpcom_remote_login_remove_dom_node_id( "wpcom_request_access_styles" );
}

// Watch for messages back from the remote login
window.addEventListener( "message", function( e ) {
	if ( e.origin === "https://r-login.wordpress.com" ) {
		var data = {};
		try {
			data = JSON.parse( e.data );
		} catch( e ) {
			wpcom_remote_login_final_cleanup();
			return;
		}

		if ( data.msg === 'LOGIN' ) {
			// Clean up the login check iframe
			wpcom_remote_login_remove_dom_node_id( "wpcom_remote_login_key" );

			var id_regex = new RegExp( /^[0-9]+$/ );
			var token_regex = new RegExp( /^.*|.*|.*$/ );
			if (
				token_regex.test( data.token )
				&& id_regex.test( data.wpcomid )
			) {
				// We have everything we need to ask for a login
				var script = document.createElement( "script" );
				script.setAttribute( "id", "wpcom_remote_login_validate" );
				script.src = '/remote-login.php?wpcom_remote_login=validate'
					+ '&wpcomid=' + data.wpcomid
					+ '&token=' + encodeURIComponent( data.token )
					+ '&host=' + window.location.protocol
						+ '//' + window.location.hostname;
				document.body.appendChild( script );
			}

			return;
		}

		// Safari ITP, not logged in, so redirect
		if ( data.msg === 'LOGIN-REDIRECT' ) {
			window.location = 'https://wordpress.com/log-in?redirect_to=' + window.location.href;
			return;
		}

		// Safari ITP, storage access failed, remove the request
		if ( data.msg === 'LOGIN-REMOVE' ) {
			var css_zap = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } @media screen and ( max-width: 782px ) { html { margin-top: 0 !important; } * html body { margin-top: 0 !important; } }';
			var style_zap = document.createElement( 'style' );
			style_zap.type = 'text/css';
			style_zap.appendChild( document.createTextNode( css_zap ) );
			document.body.appendChild( style_zap );

			var e = document.getElementById( 'wpcom_request_access_iframe' );
			e.parentNode.removeChild( e );

			document.cookie = 'wordpress_com_login_access=denied; path=/; max-age=31536000';

			return;
		}

		// Safari ITP
		if ( data.msg === 'REQUEST_ACCESS' ) {
			console.log( 'request access: safari' );

			// Check ITP iframe enable/disable knob
			if ( wpcom_remote_login_extra_auth !== 'safari_itp_iframe' ) {
				return;
			}

			// If we are in a "private window" there is no ITP.
			var private_window = false;
			try {
				var opendb = window.openDatabase( null, null, null, null );
			} catch( e ) {
				private_window = true;
			}

			if ( private_window ) {
				console.log( 'private window' );
				return;
			}

			var iframe = document.createElement( 'iframe' );
			iframe.id = 'wpcom_request_access_iframe';
			iframe.setAttribute( 'scrolling', 'no' );
			iframe.setAttribute( 'sandbox', 'allow-storage-access-by-user-activation allow-scripts allow-same-origin allow-top-navigation-by-user-activation' );
			iframe.src = 'https://r-login.wordpress.com/remote-login.php?wpcom_remote_login=request_access&origin=' + encodeURIComponent( data.origin ) + '&wpcomid=' + encodeURIComponent( data.wpcomid );

			var css = 'html { -webkit-transition: margin-top 1s; transition: margin-top 1s; } /* 9001 */ html { margin-top: 46px !important; } * html body { margin-top: 46px !important; } @media screen and ( max-width: 660px ) { html { margin-top: 71px !important; } * html body { margin-top: 71px !important; } #wpcom_request_access_iframe { display: block; height: 71px !important; } } #wpcom_request_access_iframe { border: 0px; height: 46px; position: fixed; top: 0; left: 0; width: 100%; min-width: 100%; z-index: 99999; background: #23282d; } ';

			var style = document.createElement( 'style' );
			style.type = 'text/css';
			style.id = 'wpcom_request_access_styles';
			style.appendChild( document.createTextNode( css ) );
			document.body.appendChild( style );

			document.body.appendChild( iframe );
		}

		if ( data.msg === 'DONE' ) {
			wpcom_remote_login_final_cleanup();
		}
	}
}, false );

		<h1 class="entry-title">1.01 C</h1>

				<div class="entry-meta">
						<span class="sep">Posted on </span><a href="https://wanderinginn.com/2017/04/26/1-01-c/" title="7:24 pm" rel="bookmark"><time class="entry-date" datetime="2017-04-26T19:24:08+00:00">April 26, 2017</time></a><span class="by-author"> <span class="sep"> by </span> <span class="author vcard"><a class="url fn n" href="https://wanderinginn.com/author/pirateaba/" title="View all posts by pirateaba" rel="author">pirateaba</a></span></span>		</div><!-- .entry-meta -->

	<div class="entry-content">
		<p>Perhaps it’s just as well the gods are dead. Because as I hold a dying child in my arms, I can’t believe in anything greater. Not God, not heaven—</p>
<p>Certainly not myself.</p>
<p>I’ve always been a failure. It eats at me in the dark times and my happy moments. Before, I had medicine to stop the thoughts clawing at me, dragging me down. But now I see I was always right. The happy times were an illusion.</p>
<p>I can do nothing. I am nothing. I can’t be a hero, I can’t fight. I can’t even save a single kid.</p>
<p>Wilen gasps as he clings to me. He’s still alive. But not for long. I can see it. His spine and lower torso are—</p>
<p>“It hurts. A bit.”</p>
<p>He’s biting down so hard his teeth have gone through his <em>lips</em>. I hold him to my chest. I don’t know what to say.</p>
<p>“You’re going to be okay.”</p>
<p>He gives me the same look as whenever I make a truly terrible joke. It almost makes me want to laugh.</p>
<p>“I’m going to die. But you should run. He’s coming.”</p>
<p>I can hear the Troll thumping towards me down the street. The gargantuan bastard is taking his time, strolling along. I don’t move. I can’t.</p>
<p>“I’m not leaving you here.”</p>
<p>“You’ve got to. I am dying.”</p>
<p>He says it so matter-of-fact, but his small hand is squeezing my arm. I know he’s right. He does too. I don’t know what to do. He holds something out to me. A sword? No—just a knife. A kitchen knife. He must have tried to defend himself.</p>
<p>It falls from his fingers. He can’t even hold it. Wilen is in pain. He’s—I can see the hurt in his eyes.</p>
<p>So I tell a joke.</p>
<p>“Hey. I’ve got a joke for you.”</p>
<p>I say it out of desperation, out of fear. Because he hurts and I want to take away the pain.</p>
<p>Wilen looks at me. He tries to smile and fails. Sweat and tears roll down his face, indistinguishable.</p>
<p>“Is it a good one?”</p>
<p>“You tell me.”</p>
<p>“Okay.”</p>
<p>I take a deep breath. I don’t have a joke. But it comes to me in a flash.</p>
<p>“Wh-what do you call a fat [Clown] sitting in front of a Troll?”</p>
<p>Wilen gasps. His face goes a little paler, and he grips my arm as strongly as he can. His grasp is weakening. I look down, and see the blood pooling under the skin.</p>
<p>“What? What’s a…fat [Clown] sitting in front of a Troll?”</p>
<p>“The main course.”</p>
<p>It’s not funny. But he giggles, and blood trickles out of his stomach.</p>
<p>“That’s good. But I don’t want—”</p>
<p>He loses focus for a second and stops speaking. I hold him in my arms for a while, waiting for him to finish the sentence. Then I realize he’s dead.</p>
<p>Behind me the earth shakes and I hear a heavy laugh from overhead. I smell a terrible, putrid stench, and the smell of blood and death.</p>
<p>But I don’t care.</p>
<p>My head is swimming, but my eyes are dry. All my tears run from my heart. Like blood, dripping onto the ground. My blood.</p>
<p>Everything is spinning. I can feel death approaching, snuffling and grunting. The Troll.</p>
<p>I close Wilen’s eyes and move. I stand up. I keep moving. And I smile.</p>
<p>I am not happy. I don’t think I’ll ever be happy again. But this is tragedy, and I can either fold up and die, or move on.</p>
<p>Because that’s what a clown does. I think I get it. We make comedy out of tragedy, and pretend everything’s funny when it’s not. We…</p>
<p>Tell jokes so no one can see the tears inside.</p>
<p>The Troll is behind me. I see his massive shadow as he raises his club. Laughing. But he doesn’t get it. He doesn’t understand true humor. He doesn’t understand anything.</p>
<p>I close my eyes. Sometimes you just need to let go. And let the funny inside you out.</p>
<p><em>Ha. It was a pretty funny joke.</em></p>
<p><em>Hahahaha.</em></p>
<p><em> </em></p>
<p><span style="font-size:200%;"><em>HAHAHAHAHAHAHAHAHAHAHAHAHAHAHAHAHAHAHAHAHA—</em></span></p>
<p>&nbsp;</p>
<p>[Clown Level 24!]</p>
<p>[Skill – Greater Pain Tolerance Obtained.]</p>
<p>[Skill – Devil’s Luck Obtained.]</p>
<p>&nbsp;</p>
<p><span style="color:#ff0000;">[Condition – Horrific Laughter Received.]</span></p>
<p><span style="color:#ff0000;">[Condition – Lesser Insanity Received.]</span></p>
<p>&nbsp;</p>
<p>Oh. <em>Oh.</em></p>
<p>So <em>that’s </em>what it feels like. It feels great. It feels aMaZIng.</p>
<p>I was right all along. Classes are what we make them. A [Clown] might not exist in this world, but we brought a bit of our world with us.</p>
<p>And [Clowns] aren’t just funny.</p>
<p>We’re<em> hilarious.</em></p>
<p>Remember when I said I had depression? I lied.</p>
<p>&nbsp;</p>
<p><img data-attachment-id="1760" data-permalink="https://wanderinginn.com/2017/04/26/1-01-c/reveal/" data-orig-file="https://wanderinginn.files.wordpress.com/2017/04/reveal1.png" data-orig-size="517,73" data-comments-opened="1" data-image-meta="{&quot;aperture&quot;:&quot;0&quot;,&quot;credit&quot;:&quot;&quot;,&quot;camera&quot;:&quot;&quot;,&quot;caption&quot;:&quot;&quot;,&quot;created_timestamp&quot;:&quot;0&quot;,&quot;copyright&quot;:&quot;&quot;,&quot;focal_length&quot;:&quot;0&quot;,&quot;iso&quot;:&quot;0&quot;,&quot;shutter_speed&quot;:&quot;0&quot;,&quot;title&quot;:&quot;&quot;,&quot;orientation&quot;:&quot;0&quot;}" data-image-title="Reveal" data-image-description="" data-medium-file="https://wanderinginn.files.wordpress.com/2017/04/reveal1.png?w=300" data-large-file="https://wanderinginn.files.wordpress.com/2017/04/reveal1.png?w=517" class=" size-full wp-image-1760 aligncenter" src="https://wanderinginn.files.wordpress.com/2017/04/reveal1.png?w=584" alt="Reveal" srcset="https://wanderinginn.files.wordpress.com/2017/04/reveal1.png 517w, https://wanderinginn.files.wordpress.com/2017/04/reveal1.png?w=150 150w, https://wanderinginn.files.wordpress.com/2017/04/reveal1.png?w=300 300w" sizes="(max-width: 517px) 100vw, 517px"   /></p>
<p>&nbsp;</p>
<p>And right now, the world is terribly, awfully, and horribly funny to me.</p>
<p>I put down the small body on the ground and stand up. The Troll stands over me, licking his lips. He laughs thickly, cruelly. That’s not a laugh. I show him what a <em>real</em> laugh sounds like.</p>
<p>“Hehe. Ohoho. Hahahaha. <em>HAHAHAHAHAHAHAHAHAHA.</em>”</p>
<p>It hears me laughing. The Troll’s huge face wrinkles up in confusion. He doesn’t like what he hears. He doesn’t like the sound I’m making.</p>
<p>He swings his club. I go flying. Like a bird! I’m just a clown. I can’t dodge. Why would anyone think I could dodge?</p>
<p>I think my bones break as I hit the ground. But I sit up and keep laughing. Pain is just another side of pleasure, right? It’s so funny my broken ribs don’t even bother me.</p>
<p>The Troll stomps over towards me. He punches me as I sit up and down I go. Whee!</p>
<p>Haha. It hurts! It hurts and I bleed.</p>
<p>The Troll reaches to grab me, and then pauses as he notices the knife in his arm. It goes in and out as I stab.</p>
<p>“Stab. Stab. Stab. Stabbity.”</p>
<p>He stares at me. I <em>giggle</em>. Trolls bleed red! A wonderful discovery!</p>
<p>The Troll <em>roars </em>in my face. Ooh, scary. He steps back and raises his club. A bit too far!</p>
<p>“Aha! A weak spot!”</p>
<p>This time he <em>screams</em> as I stick the knife right into his loincloth. He throws me away and I roll around. Poor bones! Poor flesh! Where’s my healing potion?</p>
<p>The Troll stares at me as I lurch to my feet. He clutches at his groin and staggers away. I’m not hurting him much, but he’s too afraid. Of what? My face? The laughing? Can’t he take a joke?</p>
<p>I take two steps and remember. Oh right. Bones are broken! I take the potion Richard gave me. How horrible it is! What’s the point of getting hurt if you’re going to fix it so easily?</p>
<p>I drink the potion down completely. And then a second. It burns like fire. Healing potions aren’t meant to be consumed so much as applied. Too much, and you’ll suffer. But what’s suffering?</p>
<p>The Troll runs off—well, limps, really. I follow him. How am I supposed to find anyone with all this <em>smoke?</em> I keep tripping over dead bodies. Honestly. You’d think the villagers would clean up after themselves after they go bleeding all over the place.</p>
<p>That’s when I see the Demons. Eight of them, standing together. A group of archers, firing at someone in the distance. I hear a <em>scream</em>, and then the Troll shouts something at the archers as he runs towards them.</p>
<p>One turns her face, and I see a harsh, angular face, dark purple fur, and two ram’s horns. Hey, Eddy was right! They do sort of look like that.</p>
<p>“Wow. She is <em>hot!</em>”</p>
<p>But she had compound eyes, and some kind of chitin on her upper body that looks like armor. Mutants. Oh god.</p>
<p>That’s too funny. I’m fighting <em>X-men?</em></p>
<p>“Hah!”</p>
<p>She looks at me. She’s probably in charge of this merry band. And oh look, they’ve got bows!</p>
<p>The other demons vary in appearance as well. One looks like he’s got fish scales, another one looks human except for the horn sticking out of his forehead, and another one is sliming from the arm.</p>
<p>Something inside of me wants to rage out. Something in me is disappointed. Demons? These are just people with skin problems. They’re no different than humans. They work as a team—they fight and kill, and they probably fear and hate us as much as we hate them.</p>
<p>What a crappy thing to be afraid of.</p>
<p>But that’s not what a [Clown] does! My thoughts twist, and my lips do the same. A [Clown] points out the way the world is broken and laughs at it. I’m an entertainer. A funny guy.</p>
<p>“Smile, smile, <em>smile.</em>”</p>
<p>The Troll runs past the group of archers, ignoring their leader’s orders to stop or take a bath. And now they’re looking around. Oh my. Did I scare the big bad demons?</p>
<p>One of them looks my way and shouts something. The others raise their bows and take aim, but uncertainly. Oops. I’m in the shadows, aren’t I? Well, time for the dramatic reveal.</p>
<p>I step out and give them a big grin and a bow. First impressions are important. And I must be a good showman.</p>
<p>Oddly, they all freeze when my face comes into view. Why? Is it my winning smile? Something on my face?</p>
<p>“Don’t look so frightened. I’m here to put a smile on your faces. Want to hear a funny joke?”</p>
<p>The Demon lieutenant raises her bow. She looses an arrow at my head before I can blink.</p>
<p>Something rips open my cheek. I blink, and touch at the cut flesh.</p>
<p>“Hey! I nearly died!”</p>
<p>She blinks at me. I glare at her, and then grin as I lick the blood.</p>
<p>“What do you call a fat man standing in front of a bunch of archers? A pincushion?”</p>
<p>I laugh and the Demons stare at me uncertainly. Don’t they get the <em>joke</em>? Or can’t they hear me? That’s even funnier.</p>
<p>“<em>HAHAHAHAHAHAHAHAHA—</em>”</p>
<p>Another arrow misses my head as I double over, full of mirth. Whoops! I stand back upright, and spread my arms out. I walk towards them, shouting merrily.</p>
<p>“Come on then! Fire! Fill me full of arrows!”</p>
<p>The archers hesitate. They’ve never seen anyone just walk at them, arms stretched out wide. I grin at the gleaming arrow tips.</p>
<p>One of them fires. This time he hits me in the leg. I stumble, but hey, what’s an arrow between friends? There’s a potion in my pocket. I pull it out of the air and drink it as another one hits my other leg. And I move towards them.</p>
<p>“Do you really think—”</p>
<p>Another arrow appears in my chest and I stumble. But the potion works. I regain my footing and keep walking.</p>
<p>“Do you really think you can shoot a guy with a great smile like mine?”</p>
<p>The arrow tips are seeking my heart. But they’ve missed. And there’s nothing for them to pierce, anyways. My heart’s in my mouth. Get it? Ha.</p>
<p>One of the leather-clad Demons lowers his bow as I draw close to him. He rips a sword out of his sheathe. Ooh, <em>scary.</em></p>
<p>I wave at him, and smile wider. His eyes are wide as he stares at me. What? Am I <em>that</em> ugly or is it the lack of horns?</p>
<p>“Hey. Wanna see a magic trick?”</p>
<p>He blinks at the unfamiliar words. But that’s okay; I’m going to show him. I flick my hand, and a knife appears in it. I’d prefer a bunny, but they’re harder to kill people with.</p>
<p>The archer isn’t ready for that. He hesitates, and I slash at his throat.</p>
<p>I miss. He’s got a Skill, or maybe he’s just more athletic than an uncoordinated guy with arrows sticking out his chest. He leans back and the knife strikes his armor instead. The impact jars the blade out of my hand.</p>
<p>Oops. He stabs towards me with a sword, and I grab his hand. He shakes me off and tries to slash at me, but suddenly he’s not holding anything. His sword is <em>gone</em>. I waggle my fingers at him.</p>
<p>“<em>Tada</em>.”</p>
<p>Then I pull it back out and run him through with it. He chokes, and I laugh as I push him away.</p>
<p>“Now, anyone know where I can find a pencil?”</p>
<p>I hear a shout, and turn my head. An arrow spins me around. That one nearly got me for good! The Lieutenant drops his bow and charges with a long knife in hand.</p>
<p>“Aren’t you just precious? Here, catch!”</p>
<p>I toss the sword at her, and she dodges left. Right into the knife that’s suddenly in my hand.</p>
<p>But of course she’s a soldier! Big, bad, and oh-so-skillful. She deflects the knife and then rushes at me. She <em>kicks</em> me so hard I feel something burst in my gut. But the potion’s still in me, and you know what? It doesn’t hurt <em>that </em>much.</p>
<p>I fall to my knees as she raises the dagger over my head. She brings it down, and I bring my hand up. I catch the blade with my fingers.</p>
<p>“Hey, this thing’s sharp! Mind if I borrow it?”</p>
<p>The long, curved dagger disappears and the Lieutenant gasps. Probably because I just stabbed her in the stomach with her own weapon.</p>
<p>“What, no applause? Your other friend thought it was good enough to <em>die for.</em>”</p>
<p>I nudge her a few times with an elbow as she staggers back.</p>
<p>“Get it? Die for? No?”</p>
<p>The other archers charge me, screaming, trying to protect their boss.</p>
<p>“Oh come on. One at a time. Form a line!”</p>
<p>I pull an arrow out of my shoulder and slash at one of the soldiers. He dodges. Ooh, he’s got a knife too.</p>
<p>He should have stabbed me in the face or somewhere wonderfully painful then, but I grin at him as he lunges, and he goes all stiff. His eyes go wide, and I make them even wider as I stab the arrow into his right socket.</p>
<p>“Look at me! <em>Look at</em>—oh, wait. You can’t.”</p>
<p>I need better material. But now the other soldiers are pausing. Their leader is desperately trying to pull the blade out and use a potion before she bleeds out. So I give them my biggest smile to show them I don’t want to hold a grudge.</p>
<p>They run.</p>
<p>“Hey! The performance isn’t over yet!”</p>
<p>I run after them, but I’m too slow. And blood loss and all that. I look down and realize I’m still bleeding from several spots.</p>
<p>“Potions. They just don’t work like they used to, do they? Back when I was a lad, we all had negative AC scores and mages could only cast 6<sup>th</sup> level spells. Bah.”</p>
<p>The world spins, but it’s nothing compared to the twisting in my head, and the laughter in my ears. Who’s laughing?</p>
<p>Oh, right. It’s me.</p>
<p>But I have no audience to share the funny with? I look around. Nope. Just dead bodies.</p>
<p>Ooh, and one freshly dead Demon lieutenant! She’s trying to drag herself away. Looks like her healing potions aren’t as nice as mine.</p>
<p>She screams when I twist my head and torso down to stare sideways at her. What’s wrong? I even smiled.</p>
<p>“Would you like to hear a joke?”</p>
<p>She slashes at me wildly with the dagger covered in her own blood. I tut and take it away from her.</p>
<p>“Not very nice is it? Now, I’m not a good [Clown]—barely past Level 20, so give me a break, will you?”</p>
<p>I start chatting as I drag her through the dirt and blood on the street. She’s shouting something, but I’m trying to tell a joke.</p>
<p>“It’s really, really good of you to put up with my bad jokes. I’m just ripping off some of the greats, you know? Improvising my act. I’ll come up with something original soon, I promise.”</p>
<p>She spits at me. I laugh, and she tenses.</p>
<p>“Okay, okay. Too real? Too personal? I know. A [Clown] should be professional. Sorry about that. On to the real material! You wanna know how I got these scars…? Nah, just yanking your leg.”</p>
<p>I pull it and she gasps a bit and tries to get away. I kick her in her half-healed wound and she convulses. Then I pull the knife out and shove it in her mouth. Not too far, just far enough that it tickles the back of her throat. Her eyes go wide as I bring my grinning face right up to hers.</p>
<p>“Here’s the joke! All you—you people out there. You’re fighting against these humans and you think you’re the scariest things around. But here’s the funny thing: you’re just killing people like you. Ordinary people who become soldiers. We called you monsters and you believed it. But listen. You’ve never seen a real monster.”</p>
<p>I shove the knife a bit further into her mouth and she gurgles. The whites of her eyes are showing as I bring my grinning face close to hers.</p>
<p>I’m smiling. I can’t stop. Her rough skin rasps below my gloved hands as I whisper in her ears.</p>
<p>“Why so <em>serious?</em> Can’t you take a joke?”</p>
<p>That was cheap. But she still doesn’t hear me. The Demon officer shouts something as I drag her down the street. Towards a burning building. I can hear the horses screaming in their stable.</p>
<p>“Shh. Shh. Let’s go say hi to the nice horsies, shall we?”</p>
<p>There’s a heavy bar on the door. I huff and puff and nearly pull a muscle getting it out! By the time the doors open, the lieutenant is trying to run down the street.</p>
<p>“Hold on, hold on! We just got to the good part! And after you came all this way—let’s watch the final part of the show together, shall we?”</p>
<p>I tackle her to the ground and grab her by the ankle. Ooh, cloven hooves. <em>Kinky.</em></p>
<p>She kicks me a few times in the face, but I don’t leg go. I’m just watching the stables. Any second now…</p>
<p>The horses inside the stable are nearly mad from fear and panic from the flames and blood in the air. When they sense there’s a way out, they kick through their paddocks and rush towards the exit.</p>
<p>The doors fly open. The horses rush out. The Demon girl I’m holding screams and tries to get away, but I hold her in place.</p>
<p>“<em>My little pony, my little pony…</em>something, something, look out for the hooves!”</p>
<p>The maddened horses stampede past us, screaming and braying so loudly I can’t even hear the Lieutenant screaming. She doesn’t even try to move; just curls up into a ball and completely misses the entertainment!</p>
<p>Me? I’m laughing.</p>
<p>The first horse knocks me flat and I hear ringing. And laughing. Another one hits me in the face and my nose <em>breaks. </em>Another steps on my hand. The bones <em>break</em> and I feel the flesh ripping. And I’m still laughing.</p>
<p>&nbsp;</p>
<p><em>Hahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahahaha—</em></p>
<p><em> </em></p>
<p>It’s morning and the sunlight is bright on my skin before I realize she’s gone. I blink, look down, and notice the blood. It’s all over me. On my hands, my face, my clothes…</p>
<p>I’m standing in a ring of bodies. Not human bodies; Demon corpses. Soldiers. They’re all dead.</p>
<p>Except one. How do I keep missing the last guy? Must be a blind spot. Probably the eye I can’t see out of at the moment.</p>
<p>I turn towards him staggering a bit. Hey, what happened to my body? Looks like someone tap-danced on my skin with a pair of skates.</p>
<p>The Demon soldier flinches as I turn my face towards him. I try to appear composed. Shame on me. I’m at a wake, after all. His. Did he hear me chuckling?</p>
<p>He’s got a strange, half-melted face. Not from any injury—it looks too clean to be a wound. The five bloody holes in his leather armor on the other hand are probably my fault.</p>
<p>“Hey pal. You don’t look so good.”</p>
<p>The soldier bares his teeth at me and tries to raise a mace. But he’s too far gone. He says something as he collapses in a puddle of his own gore. I don’t know the words, but I understand him even so.</p>
<p>“Monster.”</p>
<p>That’s fair. Harsh, but fair. I feel at my face. Cuts—broken skin—nothing like how I remembered leaving it. But the face paint is still mostly intact, which is the important thing. Actually, I think all that blood’s started a new layer. How avant-garde!</p>
<p>I look around for my special friend, the Lieutenant who can’t take a joke. Is she this body? This one? No?</p>
<p>I turn over the corpses, but find no horns—at least, not the ones I’m looking for. She’s not among them. I can’t tell if I’m disappointed or relieved.</p>
<p>I’m aware…my thoughts aren’t quite right. I’m alive. After all of this, I’m alive. I shouldn’t be, but I’m alive.</p>
<p>And I’m still insane.</p>
<p>“Heh. Ha. Hahaha.”</p>
<p>Oops. Looks like I was completely right. But you know that thing about the hidden cost and all that? Probably should have read the fine print.</p>
<p>“Tom, Tom, lost his mind! What a shock; what a bind! <em>Ahahahahaha!</em>”</p>
<p>That’s how they find me. Laughing and covered in blood, out of my mind in the ruins of the village.</p>
<p>It’s a huge group of people holding weapons. Some of them I know; others are new faces.</p>
<p>I wave at Richard. He’s leading a group of eighteen [Soldiers] and several of the combat-capable townsfolk from a few miles down the road. He stops as I approach, and the [Soldiers] raise their weapons. The civilians just freeze and stare at me.</p>
<p>“Tom…?”</p>
<p>Richard’s dark skin is very pale behind the faceguard of his helmet. He looks at me, and I can see the whites of his eyes.</p>
<p>I look at him, and he flinches. The others can’t even look at me. I catch Emily throwing up on the ground, and even Richard’s face is far too pale. I grin at him, and he takes a step back.</p>
<p>“Hey. Want to see a magic trick?”</p>
<p>“Magic…?”</p>
<p>I spread my arms and bow theatrically. I gesture towards the death and destruction around me; the dead bodies and the burned village.</p>
<p>“<em>Tada.</em>”</p>
<p>&nbsp;</p>
<p>&#8212;-</p>
<p>&nbsp;</p>
<p>The village is destroyed. Those who didn’t flee for their lives perished at the hands of the Demons and monsters. But the Demons perished at the hands of terrible comedy, and I think the Troll just decided to go back home.</p>
<p>The people from my world wander the ruins with the few villagers, staring at the death as the soldiers collect bodies and stare at me. Actually, everyone stares at me.</p>
<p>My friends—well, the people I arrived with can’t even look at me. I give them my best smile and they just freeze up. Or run away. But the soldiers and townspeople don’t look at me the same way.</p>
<p>They can’t stand the carnage either. But they look towards me and the dead with a different perspective. They count. They see only enemy soldiers, and me standing surrounded by the dead.</p>
<p>They have something else in their eyes. Fear. Fear of the unknown. But hope, too. The funniest thing. They don’t understand what I’ve done, or what I am. But I killed monsters and they think that makes me a good person.</p>
<p>Hilarious.</p>
<p>Sleep and waking are the same part of the nightmare. I hear the voice, those dispassionate words that aren’t said so much as projected into my mind.</p>
<p><em> </em></p>
<p>[Hero Level 2!]</p>
<p>[Skill – Weapon Proficiency: Knives Obtained!]</p>
<p><em> </em></p>
<p>It’s so funny I can’t stop laughing. And crying.</p>
<p>I’m a [Clown] who kills people. I’m a fat man who’s lost his funny. I’m just a guy who kills people and lets kids die—</p>
<p>But I guess I’m still a [Hero], too. If that’s not funny, I don’t know what is.</p>
<p>Want to hear a joke? A guy gets transported to a fantasy world, becomes a [Clown]—</p>
<p>And stabs people in the face. It’s not funny I’ll grant you, but I’m working on it.</p>
<p>I’m—</p>
<p>—Already far gone. Goodbye, Wilen. The part of me that still remembers whispers it. Goodbye.</p>
<p>I’m so sorry.</p>
